package com.example;

public class Calculator {
    /**
     * 두 정수를 더한 값을 반환합니다.
     * @param a 첫 번째 정수
     * @param b 두 번째 정수
     * @return 두 정수의 합
     */
    public int add(int a, int b) {
        return a + b;
    }

    public int subtract(int a, int b) {
        // 이 메서드는 테스트되지 않음
        return a - b;
    }

    public int multiply(int a, int b) {
        return a * b;
    }
}
